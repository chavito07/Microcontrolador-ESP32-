#include <Wire.h>
#include <Adafruit_INA219.h>

// Creamos el objeto del sensor
Adafruit_INA219 ina219;

void setup(void) {
  Serial.begin(115200);
  while (!Serial) { delay(1); } // Espera a que abras el Monitor Serial

  Serial.println("Hola! Buscando el sensor INA219...");

  // Inicia la comunicación con el sensor
  if (!ina219.begin()) {
    Serial.println("No se encontró el sensor. Revisa los cables!");
    while (1) { delay(10); }
  }

  Serial.println("Sensor encontrado. Midiendo...");
}

void loop(void) {
  float voltaje = 0;
  float corriente = 0;

  // Pedirle datos al sensor
  voltaje = ina219.getBusVoltage_V();
  corriente = ina219.getCurrent_mA();

  // Mostrar los datos en la pantalla de tu PC
  Serial.print("Voltaje: "); Serial.print(voltaje); Serial.println(" V");
  Serial.print("Corriente: "); Serial.print(corriente); Serial.println(" mA");
  Serial.println("--------------------");

  delay(2000); // Espera 2 segundos para la siguiente lectura
}
