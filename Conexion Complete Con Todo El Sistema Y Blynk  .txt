#define BLYNK_TEMPLATE_ID "TMPL2KL6JR3ZP"
#define BLYNK_TEMPLATE_NAME "ESPE2U"
#define BLYNK_AUTH_TOKEN "PPceQDXe0zN-Hf7AmDBt9jIv1Iv_czip"

#include <WiFi.h>
#include <WiFiClient.h>
#include <BlynkSimpleEsp32.h>
#include <Wire.h>
#include <Adafruit_INA219.h>

char auth[] = BLYNK_AUTH_TOKEN;
char ssid[] = "PUGB-MOVILE-5G"; 
char pass[] = "Matematicas";

Adafruit_INA219 ina219;
BlynkTimer timer;

void enviarDatosSensor() {
  // 1. Lectura de todos los datos del sensor
  float voltaje = ina219.getBusVoltage_V();
  float corriente_mA = ina219.getCurrent_mA();
  float potencia_mW = ina219.getPower_mW();
  float shunt_mV = ina219.getShuntVoltage_mV();

  // 2. Enviando los 4 datos a la App de Blynk
  Blynk.virtualWrite(V0, voltaje);      // Gauge de Voltaje
  Blynk.virtualWrite(V1, corriente_mA); // Gauge de Corriente
  Blynk.virtualWrite(V2, potencia_mW);  // Gauge de Potencia (mW)
  Blynk.virtualWrite(V3, shunt_mV);     // Dato de precisión (mV)

  // 3. Monitor Serie (para que veas que todo está bien en el PC)
  Serial.print("V: "); Serial.print(voltaje);
  Serial.print("V | I: "); Serial.print(corriente_mA);
  Serial.print("mA | P: "); Serial.print(potencia_mW);
  Serial.println("mW");
}

void setup() {
  Serial.begin(115200);
  delay(1000);
  Serial.println("--- PROYECTO MONITOR SOLAR UNIMAYOR ---");

  // Iniciar sensor INA219
  if (!ina219.begin()) {
    Serial.println("Error: No se encontro el INA219. Revisa cables SDA/SCL.");
    while (1); 
  }
  Serial.println("1. Sensor INA219: OK");

  // Intento de conexión WiFi
  Serial.println("2. Intentando conectar al WiFi...");
  WiFi.begin(ssid, pass);
  
  int intentos = 0;
  while (WiFi.status() != WL_CONNECTED && intentos < 20) {
    delay(500);
    Serial.print(".");
    intentos++;
  }

  if(WiFi.status() == WL_CONNECTED){
      Serial.println("\n3. WiFi: ¡CONECTADO!");
      Blynk.config(auth);
      Blynk.connect();
      Serial.println("4. Blynk: ¡LISTO!");
  } else {
      Serial.println("\nERROR: No conectó al WiFi. Revisa el Hotspot de tu cel.");
  }

  // Ejecutar la función cada 2 segundos
  timer.setInterval(2000L, enviarDatosSensor);
}

void loop() {
  Blynk.run();
  timer.run();
}
